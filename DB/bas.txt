CREATE TABLE IF NOT EXISTS `db_version` ( `Id` INT NOT NULL AUTO_INCREMENT , `Version` INT NOT NULL , `Date` DATETIME NOT NULL , PRIMARY KEY (`Id`)); 
CREATE TABLE IF NOT EXISTS `container` (`Id` int(11) NOT NULL AUTO_INCREMENT,`Size` varchar(1) NOT NULL,PRIMARY KEY(`Id`));
CREATE TABLE IF NOT EXISTS `customers` (`Id` int(11) NOT NULL AUTO_INCREMENT,`Name` varchar(100) NOT NULL,`Adr` varchar(100) NOT NULL,PRIMARY KEY (`Id`));
CREATE TABLE IF NOT EXISTS `function` (`Id` int(11) NOT NULL AUTO_INCREMENT,`Name` varchar(25) NOT NULL,`Description` text NOT NULL,PRIMARY KEY (`Id`));
CREATE TABLE IF NOT EXISTS `leaderboard` (`K_Id` int(11) NOT NULL AUTO_INCREMENT,`M_Id` int(11) NOT NULL,`Weight` int(11) NOT NULL,`Date` date NOT NULL,`Timeperiod` smallint(6) NOT NULL),PRIMARY KEY (`K_Id`,`M_Id`,`Timeperiod`);
CREATE TABLE IF NOT EXISTS `material` (`Id` int(11) NOT NULL AUTO_INCREMENT,`Name` varchar(25) NOT NULL,PRIMARY KEY (`Id`));
CREATE TABLE IF NOT EXISTS `smart_station` (`Id` int(11) NOT NULL AUTO_INCREMENT,`K_Id` int(11) NOT NULL,`Name` varchar(25) NOT NULL,PRIMARY KEY (`Id`));
CREATE TABLE IF NOT EXISTS `container_function` (`C_Id` int(11) NOT NULL,`F_Id` int(11) NOT NULL,PRIMARY KEY (`C_Id`,`F_Id`));
CREATE TABLE IF NOT EXISTS `station_container` (`S_Id` int(11) NOT NULL,`C_Id` int(11) NOT NULL,`Qnt` int(11) NOT NULL,PRIMARY KEY (`S_Id`,`C_Id`));
CREATE TABLE IF NOT EXISTS `station_material` (`S_Id` int(11) NOT NULL,`M_Id` int(11) NOT NULL,`Weight` int(11) NOT NULL,`TimeStamp` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,PRIMARY KEY (`S_Id`,`M_Id`));
